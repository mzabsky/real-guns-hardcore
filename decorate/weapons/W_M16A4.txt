    M16B A 0 A_JumpIfNoAmmo("ReloadStart")
    M16B A 0 A_ReFire
    goto Ready+13
  ReloadStart_SOH:
    M16E A 2 A_PlaySoundEx("m4a1/clipout","SoundSlot5")
    M16E BCDEFGH 2 A_JumpIfInventory("KnifeAttack",1,"KnifeAttack")
